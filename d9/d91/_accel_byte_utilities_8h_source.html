183 template<typename CharType = TCHAR, template<typename> class PrintPolicy = TPrettyJsonPrintPolicy, typename InStructType>
184 static bool TArrayUStructToJsonString(TArray<InStructType> const& InArray
185 , FString& OutJsonString
186 , int64 CheckFlags = 0
187 , int64 SkipFlags = 0
188 , int32 Indent = 0)
189 {
190 const UStruct* StructDefinition = InStructType::StaticStruct();
191 TArray< TSharedPtr<FJsonValue> > JsonArray;
192
193 for (auto& Item : InArray)
194 {
195 const void* Struct = &Item;
196 TSharedPtr<FJsonObject> JsonObject = MakeShareable(new FJsonObject());
197 if (FJsonObjectConverter::UStructToJsonObject(StructDefinition, Struct, JsonObject.ToSharedRef(), CheckFlags, SkipFlags, nullptr))
198 {
199 TSharedRef< FJsonValueObject > JsonValue = MakeShareable(new FJsonValueObject(JsonObject));
200 JsonArray.Add(JsonValue);
201 }
202 }
203
204 TSharedRef<TJsonWriter<CharType, PrintPolicy<CharType>>> JsonWriter = TJsonWriterFactory<CharType, PrintPolicy<CharType>>::Create(&OutJsonString, Indent);
205 if (FJsonSerializer::Serialize(JsonArray, JsonWriter))
206 {
207 JsonWriter->Close();
208 return true;
209 }
210 else
211 {
212 UE_LOG(LogJson, Warning, TEXT("UStructToFormattedObjectString - Unable to write out json"));
213 JsonWriter->Close();
214 }
215
216 return false;
217 }

317 template<typename ObjectType>
318 static bool UStructArrayToJsonObjectString(TArray<ObjectType> const& Objects
319 , FString& OutString)
320 {
321 OutString.Append(TEXT("["));
322 FString JsonArrayString = TEXT("");
323 for (auto& Item : Objects)
324 {
325 FString Delimiter = JsonArrayString.IsEmpty() ? TEXT("") : TEXT(",");
326 JsonArrayString.Append(Delimiter);
327 FString JsonObjectString = TEXT("");
328 if (!FJsonObjectConverter::UStructToJsonObjectString(Item, JsonObjectString))
329 {
330 return false;
331 }
332 JsonArrayString.Append(JsonObjectString);
333 }
334 OutString.Append(JsonArrayString);
335 OutString.Append(TEXT("]"));
336 return true;
337 }
