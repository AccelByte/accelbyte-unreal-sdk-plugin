14template<typename T>
15class DBQueryTask : public FNonAbandonableTask
16{
17 friend class FAutoDeleteAsyncTask<DBQueryTask>;
18 const TSharedPtr<T> Result;
19 const AccelByte::THandler<T> OnDone;
20 const double DelayTime;
21 const double TotalTimeOut;
22
23 DBQueryTask(const TSharedPtr<T>& Result, const AccelByte::THandler<T>& OnDone, double Delay = 5.0, double TimeOut = 60.0)
24 : Result(Result)
25 , OnDone(OnDone)
26 , DelayTime(Delay)
27 , TotalTimeOut(TimeOut)
28 {}
29
30 void DoWork()
31 {
32 const double TaskInitialTime = FPlatformTime::Seconds();
33
34 while (!Result.IsValid() && ((FPlatformTime::Seconds() - TaskInitialTime) < TotalTimeOut))
35 {
36 FPlatformProcess::Sleep(DelayTime);
37 UE_LOG(LogTemp, Warning, TEXT("Result: %d"), Result.IsValid());
38 }
39
40 OnDone.ExecuteIfBound(*Result.Get());
41 }
42 FORCEINLINE TStatId GetStatId() const { RETURN_QUICK_DECLARE_CYCLE_STAT(DBQueryTask, STATGROUP_ThreadPoolAsyncTasks); }
43
44};

231 template<typename T>
232 inline void FAccelByteSQLite3::SaveItemToTable(const FString& TableName, const FString& Key, const T& DataToInsert, const THandler<bool>& OnDone)
233 {
234 CreateTable(TableName, DataToInsert.StaticStruct(), THandler<bool>::CreateLambda([this, TableName, Key, DataToInsert, OnDone](bool CreateResult)
235 {
236 if (CreateResult)
237 {
238 FAutoDeleteAsyncTask<DBQueryTask<SQLiteQueryResult>>* MyTask = new FAutoDeleteAsyncTask<DBQueryTask<SQLiteQueryResult>>(MakeShareable(USQLiteDatabase::RunQueryAndGetResults(
239 DatabaseName,
240 FString::Printf(TEXT("Select name, type FROM PRAGMA_table_info('%s') ORDER BY cid;"), *TableName)).Release()),
241 THandler<SQLiteQueryResult>::CreateLambda([this, Key, DataToInsert, TableName, OnDone](SQLiteQueryResult Result)
242 {
243
244 FSQLiteTableRowSimulator Row;
245 TSharedPtr<FJsonObject> JsonObject = FJsonObjectConverter::UStructToJsonObject(DataToInsert);
246 bool bIsSuccess = true;
247 for (auto Column : Result.Results)
248 {
249 FSQLiteTableField Field = JsonObjectToTableField(JsonObject, Column.Fields[0].StringValue, Column.Fields[1].StringValue);
250 if (Field.FieldName.IsEmpty())
251 {
252 bIsSuccess = false;
253 break;
254 }
255 if (Field.FieldName == TEXT("'Key'"))
256 {
257 Field.FieldValue = FString::Printf(TEXT("\"%s\""), *Key);
258 }
259 Row.rowsOfFields.Add(Field);
260 }
261 if (bIsSuccess)
262 {
263 USQLiteDatabase::InsertRowsIntoTableUpsert(DatabaseName, TableName, TArray<FSQLiteTableRowSimulator>{ { Row }}, TEXT("Key"));
264 }
265 OnDone.ExecuteIfBound(bIsSuccess);
266 }));
267 MyTask->StartBackgroundTask();
268 }
269 else
270 {
271 OnDone.ExecuteIfBound(CreateResult);
272 }
273 }));
274 }

276 template<typename T>
277 inline void FAccelByteSQLite3::GetItemsFromTable(const FString& TableName, const THandler<TMap<FString, T>>& OnDone, int Limit, int Offset)
278 {
279 FSQLiteDatabaseReference DBRef{ DatabaseName, {TableName} };
280
281 FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>* MyTask = new FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>(MakeShared<FSQLiteQueryResult>(USQLiteDatabase::GetDataBP(
282 DBRef,
283 { "*" },
284 FSQLiteQueryFinalizedQuery{},
285 Limit,
286 Offset)),
287 THandler<FSQLiteQueryResult>::CreateLambda([this, OnDone](FSQLiteQueryResult Result)
288 {
289 if (Result.Success)
290 {
291 TMap<FString, T> Data;
292 for (int i = 0; i < Result.ResultRows.Num(); i++)
293 {
294 FString Key;
295 FString JsonString = TEXT("{");
296 bool bIsNotFirstField = false;
297 for (int j = 0; j < Result.ResultRows[i].Fields.Num(); j++)
298 {
299 if (Result.ResultRows[i].Fields[j].Key == TEXT("Key"))
300 {
301 Key = Result.ResultRows[i].Fields[j].Value;
302 Key.TrimQuotesInline();
303 continue;
304 }
305 if (bIsNotFirstField)
306 {
307 JsonString += TEXT(", ");
308 }
309 else
310 {
311 bIsNotFirstField = true;
312 }
313 JsonString += FString::Printf(TEXT("\"%s\" : %s"), *Result.ResultRows[i].Fields[j].Key, *Result.ResultRows[i].Fields[j].Value);
314 }
315 JsonString += TEXT("}");
316 typename std::remove_const<typename std::remove_reference<T>::type>::type UStructData;
317 FJsonObjectConverter::JsonObjectStringToUStruct(JsonString, &UStructData, 0, 0);
318 Data.Add(Key, UStructData);
319 }
320 OnDone.ExecuteIfBound(Data);
321 }
322 else
323 {
324 OnDone.ExecuteIfBound(TMap<FString, T>{});
325 }
326 }));
327 MyTask->StartBackgroundTask();
328 }

331 inline void FAccelByteSQLite3::GetItemFromTable(const FString& TableName, const FString& Key, const THandler<TPair<FString, T>>& OnDone)
332 {
333 FSQLiteDatabaseReference DBRef{ DatabaseName, {TableName} };
334 const FString Condition = FString::Printf(TEXT("Key = '\"%s\"'"), *Key);
335
336 FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>* MyTask = new FAutoDeleteAsyncTask<DBQueryTask<FSQLiteQueryResult>>(MakeShared<FSQLiteQueryResult>(USQLiteDatabase::GetDataBP(
337 DBRef,
338 { "*" },
339 FSQLiteQueryFinalizedQuery{ Condition })),
340 THandler<FSQLiteQueryResult>::CreateLambda([this, Key, OnDone](FSQLiteQueryResult Result)
341 {
342 if (Result.Success)
343 {
344 TPair<FString, T> Data{};
345 for (int i = 0; i < Result.ResultRows.Num(); i++)
346 {
347 FString JsonString = TEXT("{");
348 bool bIsNotFirstField = false;
349 bool bKeyFound = false;
350 for (int j = 0; j < Result.ResultRows[i].Fields.Num(); j++)
351 {
352 if (Result.ResultRows[i].Fields[j].Key == TEXT("Key"))
353 {
354 if (Result.ResultRows[i].Fields[j].Value.TrimQuotes() == Key)
355 {
356 bKeyFound = true;
357 }
358 else
359 {
360 break;
361 }
362 continue;
363 }
364 if (bIsNotFirstField)
365 {
366 JsonString += TEXT(", ");
367 }
368 else
369 {
370 bIsNotFirstField = true;
371 }
372 JsonString += FString::Printf(TEXT("\"%s\" : %s"), *Result.ResultRows[i].Fields[j].Key, *Result.ResultRows[i].Fields[j].Value);
373 }
374 JsonString += TEXT("}");
375 if (bKeyFound)
376 {
377 typename std::remove_const<typename std::remove_reference<T>::type>::type UStructData;
378 FJsonObjectConverter::JsonObjectStringToUStruct(JsonString, &UStructData, 0, 0);
379 Data = TPair<FString, T>{ Key, UStructData };
380 break;
381 }
382 }
383 OnDone.ExecuteIfBound(Data);
384 }
385 else
386 {
387 OnDone.ExecuteIfBound(TPair<FString, T>{});
388 }
389 }));
390 MyTask->StartBackgroundTask();
391 }
