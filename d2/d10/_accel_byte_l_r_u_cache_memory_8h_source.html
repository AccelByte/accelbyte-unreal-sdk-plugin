68 if (Left >= Required && bChunkCountIsSafe)

73 auto Node = this->DLLGetTail();
74 while (Node != nullptr && (Required > Left || this->ArrayGetNum() >= MemoryParameter.ChunkCount))
75 {
76 size_t TailSize = Node->GetValue().Length;
77 this->Remove(this->DLLGetTail()->GetValue().Key);
78 Left += TailSize;
79 Node = this->DLLGetTail();
80 }
