664 , FString& OutputString

666 {
667 TSharedPtr<FJsonObject> JsonObjectPtr;
668 SerializeAndRemoveEmptyValues(Model, JsonObjectPtr, bIncludeTeams);

671 FJsonSerializer::Serialize(JsonObjectPtr.ToSharedRef(), Writer);

673
674 template <typename DataStruct>
675 static void SerializeAndRemoveEmptyValues(DataStruct const& Model
676 , TSharedPtr<FJsonObject>& OutJsonObjectPtr
677 , bool bIncludeTeams = true)
678 {
679 auto JsonObjectPtr = FJsonObjectConverter::UStructToJsonObject(Model);
680
681 RemoveEmptyEnumValue(JsonObjectPtr, TEXT("joinability"));
682 RemoveEmptyEnumValue(JsonObjectPtr, TEXT("type"));
683 RemoveEmptyEnumValuesFromChildren(JsonObjectPtr, TEXT("members"), TEXT("status"));
684
685 TArray<FString> ExcludedFieldNamesForRemoval;
686 if(bIncludeTeams)
687 {
688 ExcludedFieldNamesForRemoval.Add(TEXT("teams"));
689 }
690
691 FAccelByteUtilities::RemoveEmptyFieldsFromJson(JsonObjectPtr, FAccelByteUtilities::FieldRemovalFlagAll, ExcludedFieldNamesForRemoval);
692
693 OutJsonObjectPtr = JsonObjectPtr;
694 }
