10template <class CharType>
11struct TAccelBytePrettyJsonPrintPolicy
12 : public TPrettyJsonPrintPolicy<CharType>
13{
14#if ENGINE_MAJOR_VERSION >= 5
15 static void WriteString(FArchive* Stream, const FString& String)
16 {
17 FString NewString = String;
18 FAccelByteUtilities::ReplaceDecimalSeparator(NewString, TEXT(","), TEXT("."));
19 auto Conv = StringCast<CharType>(*NewString, NewString.Len());
20 Stream->Serialize((void*)Conv.Get(), Conv.Length() * sizeof(CharType));
21 }
22
23 static void WriteFloat(FArchive* Stream, float Value)
24 {
25 WriteString(Stream, FString::Printf(TEXT("%g"), Value));
26 }
27
28 static void WriteDouble(FArchive* Stream, double Value)
29 {
30 WriteString(Stream, FString::Printf(TEXT("%.17g"), Value));
31 }
32
33
34#if ENGINE_MINOR_VERSION >= 2
35 static void WriteString(FArchive* Stream, FStringView String)
36 {
37 TempString = String.GetData();
38 FAccelByteUtilities::ReplaceDecimalSeparator(TempString, TEXT(","), TEXT("."));
39 TempStringView = TempString;
40 auto Conv = StringCast<CharType>(TempStringView.GetData(), TempStringView.Len());
41 Stream->Serialize((void*)Conv.Get(), Conv.Length() * sizeof(CharType));
42 }
43
44 static void WriteString(FArchive* Stream, FAnsiStringView String)
45 {
46 TempString = String.GetData();
47 FAccelByteUtilities::ReplaceDecimalSeparator(TempString, TEXT(","), TEXT("."));
48 TempAnsiStringView = StringCast<ANSICHAR>(*TempString);
49 auto Conv = StringCast<CharType>(TempAnsiStringView.GetData(), TempAnsiStringView.Len());
50 Stream->Serialize((void*)Conv.Get(), Conv.Length() * sizeof(CharType));
51 }
52
53 static void WriteString(FArchive* Stream, FUtf8StringView String)
54 {
55 TempString = String.GetData();
56 FAccelByteUtilities::ReplaceDecimalSeparator(TempString, TEXT(","), TEXT("."));
57 TempUtf8StringView = StringCast<UTF8CHAR>(*TempString);
58 auto Conv = StringCast<CharType>(TempUtf8StringView.GetData(), TempUtf8StringView.Len());
59 Stream->Serialize((void*)Conv.Get(), Conv.Length() * sizeof(CharType));
60 }
61
62private:
63 inline static FString TempString{};
64 inline static FStringView TempStringView{};
65 inline static FAnsiStringView TempAnsiStringView{};
66 inline static FUtf8StringView TempUtf8StringView{};
67
68#endif
69#else
70 static void WriteString(FArchive* Stream, const FString& String)
71 {
72 FString NewString = String;
73 FAccelByteUtilities::ReplaceDecimalSeparator(NewString, TEXT(","), TEXT("."));
74
75 const TCHAR* CharPtr = *NewString;
76
77 for (int32 CharIndex = 0; CharIndex < NewString.Len(); ++CharIndex, ++CharPtr)
78 {
79 TJsonPrintPolicy<CharType>::WriteChar(Stream, *CharPtr);
80 }
81 }
82#endif
83};
