140 inline bool Remove(const FName& Key)
141 {
142 int Index = FindIndex(Key);
143 auto* Node = DLLFindNode(Key);
144
145 if (Index >= 0 && Node != nullptr)

161 inline bool Emplace(const FName& Key, T& Item)
162 {
164 Remove(Key);
165
166 if (FreeCacheBeforeInsertion(Item) == false)
167 {
168 return false;
169 }
170
171 // After the existing removed, it can be added to the front
172 const FAccelByteCacheWrapper<T>* Result = InsertToCache(Item, Key);
173 if (Result == nullptr)
174 {
175 return false;
176 }
177 ArrayAdd(Result);
178 DLLAddHead(*Result);
179 return true;
180 }

190 inline TSharedPtr<T> Find(const FName& Key, bool bPeekOnly = false)
191 {
192 int Index = FindIndex(Key);
193 if (Index < 0) { return nullptr; }
194
195 if (!bPeekOnly)
196 {
197 auto* Node = DLLFindNode(Key);
198 if (Node == nullptr) { return nullptr; }
199
200 // Swap the position
201 DLLRemoveNode(Node);
202 DLLAddHead(ArrayGetIndex(Index));
203 }
204
205 return GetTheValueFromChunkArray(Index);
206 }

235 inline int FindIndex(const FName& Key)
236 {
237 int Num = ArrayGetNum();
238 for (int i = 0; i < Num; i++)

240 if (ArrayGetIndex(i).Key.IsEqual(Key, ENameCase::CaseSensitive, true))

242 return i;

254 virtual inline typename TDoubleLinkedList<FAccelByteCacheWrapper<T>>::TDoubleLinkedListNode* DLLFindNode(const FName& Key)
255 {

258 while (Node != nullptr)
259 {
260 if (Node->GetValue().Key == Key)
261 {
262 break;
263 }
264
265 Node = Node->GetNextNode();
266 }
267
268 return Node;
269 }
