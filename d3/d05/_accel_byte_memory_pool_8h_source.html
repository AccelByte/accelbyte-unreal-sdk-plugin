152 this->MemoryParameter.ChunkCount = ChunkCount_;

162 inline void RemoveAll() override

165 this->CurrentMemoryPoolSize = 0;
166 this->ChunkList.Empty();
167 }
168
169 inline const FChunkInfo<T>* Insert(T& Data, const FName& Key) override
170 {
171 if (this->ChunkList.Num() >= this->MemoryParameter.ChunkCount)

173 return nullptr;
174 }
175 // Store to the reserved memory
176 return nullptr;
177 }
178
179 inline void Remove(const FName& Key) override
180 {
181 auto Index = this->FindIndexFromChunkList(Key);
182 if (Index >= 0)
183 {
184 // deal with raw memory immediately

186 }
187 }
188
189 inline const TSharedPtr<T> Get(const FName& Key) override
190 {

195 }
196 return nullptr;
197 }
198

209 }

220template<typename T>
221class FAccelByteMemoryDynamicAllocation : public FAccelByteMemory<T>
222{
223public:
224 FAccelByteMemoryDynamicAllocation(size_t PoolSize_, int32 ChunkCount_)
225 {
226 this->MemoryParameter.Method = MemoryMethod::Dynamic;
227 this->MemoryParameter.PoolSize = PoolSize_;
228 this->MemoryParameter.ChunkCount = ChunkCount_;
229 }
230
231 ~FAccelByteMemoryDynamicAllocation()
232 {
233 this->ChunkList.Empty();
234 }
235
236 inline void RemoveAll() override
237 {
238 this->CurrentChunkCount = 0;
239 this->CurrentMemoryPoolSize = 0;
240 this->ChunkList.Empty();
241 }
242
243 inline const FChunkInfo<T>* Insert(T& Data, const FName& Key) override
244 {
245 if (!InsertPrerequisiteOkay())
246 {
247 return nullptr;
248 }
249
250 FChunkInfo<T> Result;
251 Result.Key = Key;
252 Result.Data = MakeShareable<T>(new T(Data));
253 Result.Length = FAccelByteLRUCache<T>::GetRequiredSize(Data);
254 Result.Pool = this;
255
256 int Index = this->ChunkList.Add(Result);
257 Result.Length = FAccelByteLRUCache<T>::GetRequiredSize(*this->ChunkList[Index].Data);
258
259 this->CurrentChunkCount += 1;
260 this->CurrentMemoryPoolSize += Result.Length;
261
262 return &this->ChunkList[Index];
263 }
264
265 inline void Remove(const FName& Key) override
266 {
267 auto Index = this->FindIndexFromChunkList(Key);
268 if (Index >= 0)
269 {
270 this->CurrentChunkCount -= 1;
271 this->CurrentMemoryPoolSize -= this->ChunkList[Index].Length;
272 this->ChunkList.RemoveAt(Index);
273 }
274 }
275
276 inline const TSharedPtr<T> Get(const FName& Key) override
277 {
278 auto Index = this->FindIndexFromChunkList(Key);
279 if (Index >= 0)
280 {
281 return this->ChunkList[Index].Data;
282 }
283 return nullptr;
284 }
285
286private:
287 inline bool InsertPrerequisiteOkay() override
288 {
289 if (this->GetCurrentChunkCount() >= this->MemoryParameter.ChunkCount)
290 {
291 return false;
292 }
293 if (this->GetCurrentMemoryPoolSize() >= this->MemoryParameter.PoolSize)
294 {
295 return false;
296 }
297 return true;
298 }
299};
300}
