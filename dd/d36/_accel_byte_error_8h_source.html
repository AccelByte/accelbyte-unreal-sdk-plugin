500 {
501 // custom http result for LobbyServer.GetPartyStorage
502 FString jsonString = Response == nullptr ? FAccelByteArrayByteFStringConverter::BytesToFString(Payload, true) : Response->GetContentAsString();
503 int index = jsonString.Find("\"updatedAt\"");
504 int startIndex = 0, endIndex = 0;
505
506 bool foundFirst = false;
507 bool foundLast = false;
508 for (int32 i = index; i < jsonString.Len(); i++)
509 {
510 if (FChar::IsDigit(jsonString[i]) && !foundFirst)
511 {
512 startIndex = i;
513 foundFirst = true;
514 }
515 else if (!FChar::IsDigit(jsonString[i]) && foundFirst && !foundLast)
516 {
517 endIndex = (i + 1);
518 foundLast = true;
519 }
520 }
521 jsonString.InsertAt(startIndex, "\"");
522 jsonString.InsertAt(endIndex, "\"");
523 FAccelByteModelsPartyDataNotif PartyData;
524 bool bSuccess = FAccelByteJsonConverter::JsonObjectStringToUStruct(jsonString, &PartyData);
525 if (bSuccess)
526 {
527 OnSuccess.ExecuteIfBound(PartyData);
528 }
529 return bSuccess;

546 template<typename T>
547 FHttpRequestCompleteDelegate CreateHttpResultHandler(const T& OnSuccess, const FErrorHandler& OnError, FHttpRetrySchedulerBase* Scheduler = nullptr)
548 {
549 return FHttpRequestCompleteDelegate::CreateLambda(
550 [OnSuccess, OnError, Scheduler](FHttpRequestPtr Request, FHttpResponsePtr Response, bool bFinished)
551 {
552 ACCELBYTE_SERVICE_LOGGING_HTTP_RESPONSE(Request, Response, bFinished);
553 if (Response.IsValid() && EHttpResponseCodes::IsOk(Response->GetResponseCode()))
554 {
555 if (!HandleHttpResultOk(Response, TArray<uint8>(), OnSuccess))
556 {
557 OnError.ExecuteIfBound(static_cast<int32>(ErrorCodes::InvalidResponse), "Invalid JSON response");
558 }
559 return;
560 }
561
562 if (!bFinished)
563 {
564 OnError.ExecuteIfBound(static_cast<int32>(ErrorCodes::NetworkError), "Request not sent.");
565 return;
566 }
567
568 // If response is nullptr then search the actual response in the cache
569 if (!Response.IsValid() && Scheduler != nullptr)
570 {
571 auto Cache = Scheduler->GetHttpCache().GetSerializedHttpCache(Request);
572 if (Cache.IsValid() && EHttpResponseCodes::IsOk(Cache->SerializableRequestAndResponse.ResponseCode))
573 {
574 auto ResponsePayloadByte = Cache->SerializableRequestAndResponse.ResponsePayload;
575 HandleHttpResultOk(nullptr, ResponsePayloadByte, OnSuccess);
576 return;
577 }
578 }
579
580 int32 Code;
581 FString Message;
582 HandleHttpError(Request, Response, Code, Message);
583 OnError.ExecuteIfBound(Code, Message);
584 });
585 }

593 ACCELBYTE_SERVICE_LOGGING_HTTP_RESPONSE(Request, Response, bFinished);

602

613 if (Cache.IsValid() && EHttpResponseCodes::IsOk(Cache->SerializableRequestAndResponse.ResponseCode))
614 {

616 HandleHttpResultOk(nullptr, ResponsePayloadByte, OnSuccess);

618 }
619 }

622 FString Message;
623 FJsonObject ErrorObject;
624 HandleHttpCustomError(Request, Response, Code, Message, ErrorObject);
625 OnError.ExecuteIfBound(Code, Message, ErrorObject);
